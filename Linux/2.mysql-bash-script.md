```bash
#!/bin/bash

# Configuration
DB_NAME="my_database"
DB_USER="my_user"
PASS_FILE="$HOME/.db_pass.txt"

# Generate password and save if not already saved
if [ ! -f "$PASS_FILE" ]; then
    PASSWORD=$(openssl rand -base64 16 | tr -dc 'a-zA-Z0-9' | head -c 16)
    echo "$PASSWORD" > "$PASS_FILE"
    chmod 600 "$PASS_FILE"
    echo "Password generated and saved to $PASS_FILE"
else
    PASSWORD=$(cat "$PASS_FILE")
    echo "Using existing password from $PASS_FILE"
fi

# Create the database and user using heredoc
sudo mysql -u root <<EOF
CREATE DATABASE IF NOT EXISTS \`$DB_NAME\`;
CREATE USER IF NOT EXISTS '$DB_USER'@'localhost' IDENTIFIED BY '$PASSWORD';
GRANT ALL PRIVILEGES ON \`$DB_NAME\`.* TO '$DB_USER'@'localhost';
FLUSH PRIVILEGES;
EOF

echo "Database and user setup complete."
```
